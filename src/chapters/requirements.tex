\chapter{REQUIREMENTS}
\label{chap:requirements}

\section{Hardware Requirements}

\begin{itemize}[leftmargin=1.5em]
    \item \textbf{USB Webcams $\times$ 3:} Standard 1080p USB webcams (e.g., Logitech C270 or equivalent), positioned at front, left, and right angles around the capture area. Minimum 30~fps required for real-time operation.
    \item \textbf{Camera Mounts / Tripods $\times$ 3:} Fixed mounting hardware to ensure stable, calibrated camera positions during capture sessions.
    \item \textbf{Checkerboard Calibration Target:} Printed A4 checkerboard pattern (9$\times$6 inner corners) for OpenCV camera calibration.
    \item \textbf{Laptop / Desktop PC:} Minimum Intel Core i5 (8th generation or above), 8~GB RAM. No GPU required MediaPipe and PyBullet both run on CPU for this pipeline.
    \item \textbf{USB Hub (optional):} If the host machine has insufficient USB ports for three simultaneous camera connections.
\end{itemize}

\section{Software Requirements}

\begin{itemize}[leftmargin=1.5em]
    \item \textbf{Programming Language:} Python 3.9 or above.
    \item \textbf{Pose Estimation:} MediaPipe 0.10+ (\texttt{mediapipe} Python package) - for real-time arm and hand landmark detection.
    \item \textbf{Computer Vision:} OpenCV 4.8+ (\texttt{opencv-python}) - for camera capture, calibration, and triangulation.
    \item \textbf{Physics Simulation:} PyBullet (\texttt{pybullet}) - for IK computation and real-time arm simulation rendering.
    \item \textbf{Numerical Computing:} NumPy (\texttt{numpy}) - for matrix operations in triangulation and IK.
    \item \textbf{Data Storage:} Python \texttt{csv} module or Pandas (\texttt{pandas}) - for structured dataset writing.
    \item \textbf{Development Environment:} Visual Studio Code on Windows 10/11 or Ubuntu 22.04.
    \item \textbf{Version Control:} Git / GitHub - for source code management.
\end{itemize}
